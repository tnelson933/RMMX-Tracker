on:
  push:
    tags:
      - "desktop-v*"

permissions:
  contents: write

jobs:
  # ── macOS — builds signed + notarized .dmg for x64 and arm64 ─────────────────
  build-mac:
    name: Build macOS installer
    runs-on: macos-14
    steps:
      - name: Checkout
        uses: actions/checkout@v4

      - name: Install pnpm
        uses: pnpm/action-setup@v4
        with:
          version: 10

      - name: Set up Node.js
        uses: actions/setup-node@v4
        with:
          node-version: "24"
          cache: "pnpm"

      - name: Install dependencies
        run: pnpm install --frozen-lockfile

      - name: Build local-server
        run: pnpm --filter @workspace/local-server run build

      - name: Build race-platform frontend
        run: pnpm --filter @workspace/race-platform run build
        env:
          PORT: "9090"
          BASE_PATH: "/"

      - name: Build Electron main + preload
        run: pnpm --filter @workspace/desktop run build

      - name: Import Apple certificate
        if: ${{ secrets.APPLE_CERTIFICATE != '' }}
        env:
          APPLE_CERTIFICATE: ${{ secrets.APPLE_CERTIFICATE }}
          APPLE_CERTIFICATE_PASSWORD: ${{ secrets.APPLE_CERTIFICATE_PASSWORD }}
          KEYCHAIN_PASSWORD: ${{ github.run_id }}
        run: |
          KEYCHAIN_PATH="$RUNNER_TEMP/build.keychain"
          CERT_PATH="$RUNNER_TEMP/certificate.p12"
          echo "$APPLE_CERTIFICATE" | base64 --decode > "$CERT_PATH"
          security create-keychain -p "$KEYCHAIN_PASSWORD" "$KEYCHAIN_PATH"
          security set-keychain-settings -lut 21600 "$KEYCHAIN_PATH"
          security unlock-keychain -p "$KEYCHAIN_PASSWORD" "$KEYCHAIN_PATH"
          security import "$CERT_PATH" \
            -k "$KEYCHAIN_PATH" \
            -P "$APPLE_CERTIFICATE_PASSWORD" \
            -T /usr/bin/codesign
          security list-keychains -d user -s "$KEYCHAIN_PATH"
          security set-key-partition-list \
            -S apple-tool:,apple: \
            -s -k "$KEYCHAIN_PASSWORD" "$KEYCHAIN_PATH"
          echo "KEYCHAIN_PATH=$KEYCHAIN_PATH" >> "$GITHUB_ENV"
      - name: Package macOS installer
        run: pnpm --filter @workspace/desktop exec electron-builder --mac --x64 --arm64 --publish always
        env:
          GH_TOKEN: ${{ secrets.GITHUB_TOKEN }}
          CSC_KEYCHAIN: ${{ env.KEYCHAIN_PATH }}
          CSC_LINK: ${{ secrets.APPLE_CERTIFICATE }}
          CSC_KEY_PASSWORD: ${{ secrets.APPLE_CERTIFICATE_PASSWORD }}
          APPLE_ID: ${{ secrets.APPLE_ID }}
          APPLE_TEAM_ID: ${{ secrets.APPLE_TEAM_ID }}
          APPLE_APP_SPECIFIC_PASSWORD: ${{ secrets.APPLE_APP_SPECIFIC_PASSWORD }}

      - name: Clean up keychain
        if: always()
        run: |
          KEYCHAIN_PATH="$RUNNER_TEMP/build.keychain"
          if security list-keychains | grep -q "$KEYCHAIN_PATH"; then
            security delete-keychain "$KEYCHAIN_PATH"
          fi
  # ── Windows — builds (optionally signed) NSIS .exe installer ─────────────────
  build-windows:
    name: Build Windows installer
    runs-on: windows-2022
    steps:
      - name: Checkout
        uses: actions/checkout@v4

      - name: Install pnpm
        uses: pnpm/action-setup@v4
        with:
          version: 10

      - name: Set up Node.js
        uses: actions/setup-node@v4
        with:
          node-version: "24"
          cache: "pnpm"

      - name: Install dependencies
        run: pnpm install --frozen-lockfile

      - name: Build local-server
        run: pnpm --filter @workspace/local-server run build

      - name: Build race-platform frontend
        run: pnpm --filter @workspace/race-platform run build
        env:
          PORT: "9090"
          BASE_PATH: "/"

      - name: Build Electron main + preload
        run: pnpm --filter @workspace/desktop run build

      - name: Import Windows certificate
        if: ${{ secrets.WIN_CERTIFICATE != '' }}
        env:
          WIN_CERTIFICATE: ${{ secrets.WIN_CERTIFICATE }}
        shell: pwsh
        run: |
          $certBytes = [Convert]::FromBase64String($env:WIN_CERTIFICATE)
          $certPath  = Join-Path $env:RUNNER_TEMP "win-cert.pfx"
          [IO.File]::WriteAllBytes($certPath, $certBytes)
          Write-Output "WIN_CERT_PATH=$certPath" >> $env:GITHUB_ENV
      - name: Package Windows installer
        run: pnpm --filter @workspace/desktop exec electron-builder --win --x64 --publish always
        env:
          GH_TOKEN: ${{ secrets.GITHUB_TOKEN }}
          WIN_CSC_LINK: ${{ env.WIN_CERT_PATH }}
          WIN_CSC_KEY_PASSWORD: ${{ secrets.WIN_CERTIFICATE_PASSWORD }}

      - name: Clean up Windows certificate
        if: always()
        shell: pwsh
        run: |
          if ($env:WIN_CERT_PATH -and (Test-Path $env:WIN_CERT_PATH)) {
            Remove-Item $env:WIN_CERT_PATH -Force
          }
